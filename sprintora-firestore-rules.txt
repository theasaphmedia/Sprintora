rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // --- users ---
    // Any signed-in user can read any user profile (needed to look up
    // teammates by email when inviting, and to show names on the team
    // panel). Each user can create/edit their own profile document, but NOT
    // the billing/trial fields on it — those are written only by
    // server-side routes (Admin SDK, bypasses these rules entirely):
    // /api/start-trial, the Paystack webhook, and the trial-expiry cron.
    // Without this restriction, a signed-in user could open dev tools and
    // set their own plan to "starter"/"team"/"business" with a direct
    // Firestore write, or reset trialUsed to false to claim unlimited free
    // trials.
    // Signup (app/signup/page.js) does write `plan: "beta"` itself at
    // account creation — that's still allowed below, since "beta" is the
    // safe free-tier default; only "pro" and the billing/trial fields are
    // blocked from ever being client-writable.
    match /users/{uid} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == uid
                    && !request.resource.data.keys().hasAny(
                      ['paystackCustomerCode', 'paystackSubscriptionCode', 'subscriptionStatus', 'trialEndsAt', 'trialUsed']
                    )
                    && (!('plan' in request.resource.data) || request.resource.data.plan == 'beta');
      allow update: if request.auth != null && request.auth.uid == uid
                    && !request.resource.data.diff(resource.data).affectedKeys().hasAny(
                      ['plan', 'paystackCustomerCode', 'paystackSubscriptionCode', 'subscriptionStatus', 'trialEndsAt', 'trialUsed']
                    );
      allow delete: if false;
    }

    // --- invites ---
    // Only the current OWNER of a project may create an invite for it
    // (invitedBy must be their own uid, and they must own projectId —
    // otherwise anyone could invite themselves into any project and use
    // that invite to self-join below). An invite can only be read by the
    // person who sent it or the person it was sent to (matched by email
    // on their auth token). Update is allowed for either the invitee
    // (accepting, or self-marking "left" on leaving) OR the original
    // inviter (marking "removed" when they remove someone, or re-inviting
    // via setDoc on an already-existing doc, which Firestore treats as an
    // update, not a create).
    //
    // Reading/accepting an invite by email match additionally requires
    // request.auth.token.email_verified == true. Firebase does not verify
    // ownership of an email at password sign-up, so without this check
    // anyone could sign up claiming someone else's email address and
    // hijack any invite sent to it. Google sign-in already sets
    // email_verified true automatically, so this doesn't affect that flow.
    match /invites/{inviteId} {
      allow create: if request.auth != null
                    && request.resource.data.invitedBy == request.auth.uid
                    && request.auth.uid == get(/databases/$(database)/documents/projects/$(request.resource.data.projectId)).data.ownerId;
      allow read: if request.auth != null
                  && ((resource.data.email == request.auth.token.email
                       && request.auth.token.email_verified == true)
                      || resource.data.invitedBy == request.auth.uid);
      allow update: if request.auth != null
                    && (
                      (request.auth.token.email_verified == true
                       && resource.data.email == request.auth.token.email)
                      || resource.data.invitedBy == request.auth.uid
                    );
      allow delete: if false;
    }

    // Per-project member cap, tied to the project OWNER's billing plan
    // (whoever is doing the writing — e.g. someone accepting an invite —
    // doesn't have their own plan checked here, only the owner's). Mirrors
    // lib/planLimits.js's PLAN_LIMITS.*.maxMembers on the app side; if
    // those numbers ever change, update both places by hand — rules can't
    // import a JS module, so this table is duplicated, not shared, and can
    // drift if only one side gets edited.
    function memberLimitForPlan(plan) {
      return plan == 'starter' ? 5
        : plan == 'team' ? 15
        : plan == 'business' ? 40
        : 3; // 'beta' (free) and any unrecognized value
    }

    function ownerPlan(ownerId) {
      return get(/databases/$(database)/documents/users/$(ownerId)).data.get('plan', 'beta');
    }

    // --- projects ---
    match /projects/{projectId} {
      allow create: if request.auth != null
                    && request.resource.data.ownerId == request.auth.uid
                    && request.resource.data.memberIds == [request.auth.uid];

      allow read: if request.auth != null
                  && request.auth.uid in resource.data.memberIds;

      // Update is allowed for three cases only:
      // 1) the OWNER changing anything except ownerId itself (add/remove
      //    members, rename, etc. — ownership transfer isn't a feature yet,
      //    so ownerId can never move via update). If this write grows
      //    memberIds, the new size must fit the owner's plan's member cap
      //    — shrinking or unchanged size is always allowed regardless of
      //    plan, since removing people or leaving other fields alone
      //    should never be blocked by a cap.
      // 2) any current member removing ONLY themselves from memberIds
      //    (leaving the project) — no other field, and no one else's
      //    membership, may change in the same write.
      // 3) a NON-member self-joining by adding exactly themselves to
      //    memberIds, and ONLY if a real, matching, pending invite exists
      //    for this project + their email (looked up by deterministic
      //    invite doc ID, not just trusting the shape of the write), AND
      //    their auth token's email is verified (see note on /invites —
      //    prevents joining via an invite sent to an email the requester
      //    merely typed at signup without proving they own it), AND the
      //    resulting member count still fits the owner's plan's cap.
      allow update: if request.auth != null && (
        (
          request.auth.uid == resource.data.ownerId
          && request.resource.data.ownerId == resource.data.ownerId
          && (
            request.resource.data.memberIds.size() <= resource.data.memberIds.size()
            || request.resource.data.memberIds.size() <= memberLimitForPlan(ownerPlan(resource.data.ownerId))
          )
        )
        || (
          request.auth.uid in resource.data.memberIds
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['memberIds'])
          && request.resource.data.memberIds.size() == resource.data.memberIds.size() - 1
          && resource.data.memberIds.hasAll(request.resource.data.memberIds)
          && !(request.auth.uid in request.resource.data.memberIds)
        )
        || (
          !(request.auth.uid in resource.data.memberIds)
          && request.auth.token.email_verified == true
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['memberIds', 'roles'])
          && request.resource.data.memberIds.size() == resource.data.memberIds.size() + 1
          && request.resource.data.memberIds.size() <= memberLimitForPlan(ownerPlan(resource.data.ownerId))
          && request.resource.data.memberIds.hasAll(resource.data.memberIds)
          && request.auth.uid in request.resource.data.memberIds
          && exists(/databases/$(database)/documents/invites/$(projectId + '_' + request.auth.token.email))
          && get(/databases/$(database)/documents/invites/$(projectId + '_' + request.auth.token.email)).data.status == 'pending'
          && get(/databases/$(database)/documents/invites/$(projectId + '_' + request.auth.token.email)).data.email == request.auth.token.email
        )
      );

      allow delete: if request.auth != null
                    && request.auth.uid == resource.data.ownerId;

      // --- tasks (subcollection) ---
      match /tasks/{taskId} {
        allow read, create, update, delete: if request.auth != null
          && request.auth.uid in get(/databases/$(database)/documents/projects/$(projectId)).data.memberIds;

        // --- comments (subcollection of tasks) ---
        // Any project member can read and post comments. Comments are
        // immutable once posted (no update rule — not even by the author,
        // to keep this simple for now) but the author can delete their own.
        // No one else, not even the project owner, can delete someone
        // else's comment yet — that's a moderation feature to add later if
        // actually needed, not assumed now.
        match /comments/{commentId} {
          allow read: if request.auth != null
            && request.auth.uid in get(/databases/$(database)/documents/projects/$(projectId)).data.memberIds;
          allow create: if request.auth != null
            && request.auth.uid in get(/databases/$(database)/documents/projects/$(projectId)).data.memberIds
            && request.resource.data.authorId == request.auth.uid;
          allow update: if false;
          allow delete: if request.auth != null
            && resource.data.authorId == request.auth.uid;
        }

        // --- activity (subcollection of tasks) ---
        // Auto-generated system log entries ("created", "moved") written by
        // the app itself whenever a member performs the underlying action —
        // not user-authored content, so there's no per-entry ownership
        // check beyond "you're a project member," matching the trust level
        // already granted for the task actions these entries describe.
        // Immutable and undeletable by design — it's a log, not editable data.
        match /activity/{activityId} {
          allow read: if request.auth != null
            && request.auth.uid in get(/databases/$(database)/documents/projects/$(projectId)).data.memberIds;
          allow create: if request.auth != null
            && request.auth.uid in get(/databases/$(database)/documents/projects/$(projectId)).data.memberIds
            && request.resource.data.actorId == request.auth.uid;
          allow update: if false;
          allow delete: if false;
        }
      }
    }
  }
}
